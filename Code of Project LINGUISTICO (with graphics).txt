// ─────────────────────────────────────────────
//  TEXT-TO-SPEECH  (PowerShell, no sapi.h needed)
// ─────────────────────────────────────────────
static HANDLE g_proc = NULL;

static void SpeakStop()
{
    if (g_proc) {
        TerminateProcess(g_proc, 0);
        CloseHandle(g_proc);
        g_proc = NULL;
    }
}

static void Speak(const std::string& text)
{
    SpeakStop();
    std::string s;
    for (char c : text)
        s += (c == '\'') ? std::string("''") : std::string(1, c);

    std::string cmd =
        "powershell -WindowStyle Hidden -Command \""
        "Add-Type -AssemblyName System.Speech;"
        "$v=New-Object System.Speech.Synthesis.SpeechSynthesizer;"
        "$v.Rate=-1;$v.Speak('" + s + "')\"";

    STARTUPINFOA si = {};
    si.cb = sizeof(si);
    si.dwFlags = STARTF_USESHOWWINDOW;
    si.wShowWindow = SW_HIDE;
    PROCESS_INFORMATION pi = {};
    char buf[8192];
    snprintf(buf, sizeof(buf), "%s", cmd.c_str());

    if (CreateProcessA(NULL, buf, NULL, NULL, FALSE,
                       CREATE_NO_WINDOW, NULL, NULL, &si, &pi)) {
        CloseHandle(pi.hThread);
        g_proc = pi.hProcess;
    }
}

// ─────────────────────────────────────────────
//  UI SOUNDS

static Sound MakeTone(float freq, float dur, float vol)
{
    int n = (int)(44100 * dur);
    float* d = (float*)MemAlloc(n * sizeof(float));
    for (int i = 0; i < n; i++) {
        float t = (float)i / 44100;
        d[i] = sinf(2.f * PI * freq * t) * vol * (1.f - t / dur);
    }
    Wave w = { (unsigned int)n, 44100, 32, 1, d };
    Sound s = LoadSoundFromWave(w);
    MemFree(d);
    return s;
}

// ─────────────────────────────────────────────
//  FONT  –  loaded with full Unicode codepoints
//  so that Þ þ Æ æ Ƿ ƿ ð ó á é ¿ ¡ ñ ü etc.
//  all render correctly.
// ─────────────────────────────────────────────
static Font g_font = { 0 };

static void InitCustomFont()
{
    // ── Codepoint list ─────────────────────────────────────────────────────
    // We build an explicit list so Raylib's font atlas contains EVERY glyph
    // used in the project — including Old English symbols Þ þ Æ æ Ƿ ƿ ð Ð
    // and punctuation like – — • → ─ ═ that are above U+024F.
    //
    // HOW RAYLIB FONT LOADING WORKS:
    //   LoadFont()   → only bakes ASCII 32-126   (no Unicode)
    //   LoadFontEx() → bakes whatever codepoints YOU pass in
    // So we MUST use LoadFontEx with a complete list.
    // ─────────────────────────────────────────────────────────────────────

    std::vector<int> cps;
    cps.reserve(600);

    // 1. Basic ASCII printable (32-126)
    for (int i = 32; i <= 126; i++) cps.push_back(i);

    // 2. Latin-1 Supplement + Latin Extended-A/B  (covers Þþ Ææ ðÐ óáéíúñü ¿¡ etc.)
    for (int i = 0x00A0; i <= 0x024F; i++) cps.push_back(i);

    // 3. Wynn explicitly  (U+01F7 Ƿ  U+01BF ƿ — inside range above but belt+braces)
    cps.push_back(0x01F7);  // Ƿ  capital Wynn
    cps.push_back(0x01BF);  // ƿ  small   Wynn

    // 4. General Punctuation block: en-dash, em-dash, bullet, arrows (U+2013..U+2192)
    for (int i = 0x2013; i <= 0x2192; i++) cps.push_back(i);

    // 5. Box-drawing characters used in decorative lines (U+2500..U+257F)
    for (int i = 0x2500; i <= 0x257F; i++) cps.push_back(i);

    // 6. Smart quotes (sometimes in pasted text)
    int extras[] = { 0x2018, 0x2019, 0x201C, 0x201D };
    for (int cp : extras) cps.push_back(cp);

    // Remove duplicates (simple sort+unique)
    std::sort(cps.begin(), cps.end());
    cps.erase(std::unique(cps.begin(), cps.end()), cps.end());

    int glyphCount = (int)cps.size();

    // ── Font search order ──────────────────────────────────────────────────
    // LiberationSans-Regular.ttf ships with this project (fonts/ subfolder)
    // and is the ONLY font guaranteed to contain ALL of the above including
    // Ƿ/ƿ (U+01F7/U+01BF).  Windows fonts like Arial lack Wynn entirely.
    //
    // Place LiberationSans-Regular.ttf next to the .exe  OR  in fonts/
    // (both paths are tried below).  Windows system fonts are tried after
    // that as a quality fallback for the glyphs they DO cover.
    // ─────────────────────────────────────────────────────────────────────
    const char* fontPaths[] = {
        "LiberationSans-Regular.ttf",           // same folder as .exe
        "fonts/LiberationSans-Regular.ttf",     // fonts/ sub-folder
        "C:/Windows/Fonts/arial.ttf",           // Windows fallback (missing Ƿ/ƿ)
        "C:/Windows/Fonts/segoeui.ttf",
        "C:/Windows/Fonts/calibri.ttf",
        "C:/Windows/Fonts/tahoma.ttf",
        "C:/Windows/Fonts/consola.ttf",
        nullptr
    };

    g_font.texture.id = 0;
    for (int k = 0; fontPaths[k] != nullptr; k++) {
        g_font = LoadFontEx(fontPaths[k], 52, cps.data(), glyphCount);
        if (g_font.texture.id > 0) break;
    }

    // Absolute fallback (no Unicode, but at least the app won't crash)
    if (g_font.texture.id == 0)
        g_font = GetFontDefault();

    // Bilinear filter keeps glyphs sharp at varying display sizes
    SetTextureFilter(g_font.texture, TEXTURE_FILTER_BILINEAR);
}

static void DrawTextClean(const char* text, int x, int y, int fontSize, Color color)
{
    // Always DrawTextEx — the built-in DrawText has no Unicode glyphs.
    // Thorn (Þ/þ), Ash (Æ/æ), Wynn (Ƿ/ƿ) etc. need the custom font atlas.
    DrawTextEx(g_font, text, (Vector2){(float)x, (float)y},
               (float)fontSize, 1.0f, color);
}

static void DrawTextCentered(const char* text, int y, int fontSize, Color color, int screenWidth)
{
    int tw = (int)MeasureTextEx(g_font, text, (float)fontSize, 1.0f).x;
    DrawTextClean(text, (screenWidth - tw) / 2, y, fontSize, color);
}

#define CText(t, y, fs, c, sw) DrawTextCentered(t, y, fs, c, sw)

static Rectangle RaylibBtn(int cx, int y, int w, int h) {
    return { (float)(cx - w/2), (float)y, (float)w, (float)h };
}
static bool Hov(Rectangle r) {
    return CheckCollisionPointRec(GetMousePosition(), r);
}
static bool Clk(Rectangle r) {
    return Hov(r) && IsMouseButtonPressed(MOUSE_LEFT_BUTTON);
}
static void DrawBtn(Rectangle r, const char* lbl, bool hov, bool sel, Color sc) {
    Color bg = sel ? sc : (hov ? C_HOVER : C_PANEL);
    Color tc = sel ? C_BG : (hov ? C_BG : C_WHITE);
    DrawRectangleRounded(r, 0.3f, 8, bg);
    DrawRectangleRoundedLines(r, 0.3f, 8, hov ? C_HOVER : C_ACCENT);
    int tw = (int)MeasureTextEx(g_font, lbl, 17, 1.0f).x;
    DrawTextClean(lbl, (int)(r.x + r.width/2 - tw/2),
                  (int)(r.y + r.height/2 - 9), 17, tc);
}
static bool DrawSpeaker(int x, int y, bool playing) {
    Rectangle r = { (float)x, (float)y, 46.f, 36.f };
    bool hov = Hov(r);
    Color bg = playing ? C_ACCENT : (hov ? (Color){60,50,100,255} : (Color){35,28,60,210});
    DrawRectangleRounded(r, 0.35f, 8, bg);
    DrawRectangleRoundedLines(r, 0.35f, 8, playing ? C_WHITE : (hov ? C_ACCENT : C_MUTED));
    Color ic = playing ? C_BG : C_WHITE;
    int cx = x+13, cy = y+18;
    DrawRectangle(cx-5, cy-5, 7, 10, ic);
    DrawTriangle({(float)(cx+2),(float)(cy-5)},
                 {(float)(cx+2),(float)(cy+5)},
                 {(float)(cx+11),(float)(cy-9)}, ic);
    DrawTriangle({(float)(cx+2),(float)(cy+5)},
                 {(float)(cx+11),(float)(cy+9)},
                 {(float)(cx+11),(float)(cy-9)}, ic);
    DrawTextClean(playing ? "))))" : "))", x+29, y+11, 11,
             playing ? C_BG : (hov ? C_ACCENT : C_MUTED));
    DrawTextClean("listen", x+3, y+40, 10, C_MUTED);
    return Clk(r);
}

// ─────────────────────────────────────────────
//  SCREEN STATES
// ─────────────────────────────────────────────
enum AppScreen {
    S_SPLASH, S_LANG, S_PROF,
    S_INTRO,  S_QUIZ, S_TYPE,  S_CONGRATS
};

// ─────────────────────────────────────────────
//  MAIN
// ─────────────────────────────────────────────
int main()
{
    const int SW = 980, SH = 660;
    SetExitKey(KEY_NULL);
    InitWindow(SW, SH, "Linguistico");
    SetTargetFPS(60);
    InitAudioDevice();
    InitCustomFont();

    Sound sClick   = MakeTone(660.f, 0.07f, 0.22f);
    Sound sCorrect = MakeTone(880.f, 0.25f, 0.32f);
    Sound sWrong   = MakeTone(200.f, 0.32f, 0.32f);
    Sound sIntro   = MakeTone(528.f, 0.38f, 0.22f);

    auto courses = BuildCourses();

    // ── State ────────────────────────────────
    AppScreen scr = S_SPLASH;
    int ci=0, pi=0, li=0, qi=0, score=0, totalQ=0;
    float timer=0.f, feed=0.f;
    bool answered=false, correct=false, speaking=false;
    float spkTimer=0.f, spkDur=0.f;
    std::vector<bool> sel;
    bool introReady = false;

    // Type-answer state
    char typeInput[256] = {};
    int  typeLen = 0;
    bool typeCorrect = false;
    bool typeSubmitted = false;

    // Star field
    const int NS = 100;
    float sx[100], sy[100], ss[100], sr[100];
    for (int i = 0; i < NS; i++) {
        sx[i] = (float)GetRandomValue(0, SW);
        sy[i] = (float)GetRandomValue(0, SH);
        ss[i] = GetRandomValue(1, 4) * 0.3f;
        sr[i] = GetRandomValue(1, 3) * 0.5f;
    }

    // ── Helpers ──────────────────────────────
    auto getLevel = [&]() -> const Level& {
        return courses[ci].profs[pi].levels[li];
    };
    auto getQ = [&]() -> const Question& {
        return getLevel().questions[qi];
    };
    auto resetQ = [&]() {
        answered=false; correct=false; feed=0.f; speaking=false;
        typeSubmitted=false; typeCorrect=false;
        memset(typeInput, 0, sizeof(typeInput)); typeLen=0;
        SpeakStop();
        sel.assign(getQ().options.size(), false);
    };
    auto startSpeak = [&](const std::string& txt) {
        Speak(txt);
        speaking  = true;
        spkDur    = (float)txt.size() * 0.075f + 0.6f;
        spkTimer  = 0.f;
    };
    auto goQuiz = [&]() {
        if (li >= (int)courses[ci].profs[pi].levels.size()) return;
        if (qi >= (int)getLevel().questions.size()) qi = 0;
        if (getQ().typeAnswer) scr = S_TYPE;
        else                   scr = S_QUIZ;
        introReady = false;
        resetQ();
        const std::string& st = getQ().speakText;
        startSpeak(st.empty() ? getQ().prompt : st);
    };
    auto goIntro = [&]() {
        scr = S_INTRO;
        timer = 0.f;
        introReady = false;
        PlaySound(sIntro);
        Speak("Level " + std::to_string(li+1) + ". " + getLevel().subtitle + ".");
    };
    auto startCourse = [&]() {
        li=0; qi=0; score=0; totalQ=0;
        for (auto& lv : courses[ci].profs[pi].levels)
            totalQ += (int)lv.questions.size();
        goIntro();
    };
    auto nextQ = [&]() {
        qi++;
        if (qi >= (int)getLevel().questions.size()) {
            li++;
            qi = 0;
            if (li >= (int)courses[ci].profs[pi].levels.size()) {
                scr = S_CONGRATS;
                introReady = false;
                PlaySound(sIntro);
                Speak("Congratulations! You have completed this proficiency level.");
            } else {
                goIntro();
            }
        } else {
            goQuiz();
        }
    };

    // ─────────────────────────────────────────
    //  MAIN LOOP
    // ─────────────────────────────────────────
    while (!WindowShouldClose())
    {
        float dt = GetFrameTime();
        timer += dt;
        if (feed > 0) feed -= dt * 1.5f;
        if (speaking) { spkTimer += dt; if (spkTimer >= spkDur) speaking = false; }
        for (int i = 0; i < NS; i++) { sy[i] += ss[i]; if (sy[i] > SH) sy[i] = 0; }

        BeginDrawing();
        ClearBackground(C_BG);

        // Stars
        for (int i = 0; i < NS; i++) {
            float p = 0.5f + 0.5f * sinf(timer * ss[i] + (float)i);
            DrawCircleV({sx[i], sy[i]}, sr[i],
                {(unsigned char)(180*p),(unsigned char)(130*p),255,150});
        }
        DrawRectangle(0, 0, SW, 4, C_ACCENT);

        // ══ SPLASH ═════════════════════════════
        if (scr == S_SPLASH) {
            float w = sinf(timer * 1.2f) * 5.f;
            CText("LINGUISTICO", SH/2-110+(int)w, 58, C_GOLD, SW);
            CText("A Language Learning Application", SH/2-42, 20, C_MUTED, SW);
            CText("Designed by Irtiza, Faizan, Arshad and Gohar", SH/2-14, 16, C_MUTED, SW);
            CText("~ Old English   ~   Spanish ~", SH/2+18, 22, C_ACCENT, SW);

            // Prof mode legend
            DrawRectangleRounded({(float)(SW/2-260), (float)(SH/2+52), 520.f, 36.f}, 0.3f, 8,
                                 (Color){28,22,55,200});
            DrawTextClean("Reading", SW/2-230, SH/2+62, 14, C_READ);
            DrawTextClean("(Alphabets / Basics)", SW/2-170, SH/2+62, 14, C_MUTED);
            DrawTextClean("Writing", SW/2-10, SH/2+62, 14, C_WRITE);
            DrawTextClean("(Words)", SW/2+52, SH/2+62, 14, C_MUTED);
            DrawTextClean("Translation", SW/2+120, SH/2+62, 14, C_TRANS);
            DrawTextClean("(Hard)", SW/2+215, SH/2+62, 14, C_MUTED);

            Rectangle b = RaylibBtn(SW/2, SH/2+110, 240, 54);
            DrawBtn(b, "BEGIN", Hov(b), false, C_ACCENT);
            if (Clk(b)) {
                PlaySound(sClick);
                Speak("Hello! Welcome to Linguistico. A language learning application.");
                scr = S_LANG;
            }
        }

        // ══ LANGUAGE SELECT ════════════════════
        else if (scr == S_LANG) {
            CText("Choose a Language", 75, 38, C_WHITE, SW);
            CText("Which tongue shall you learn today?", 126, 19, C_MUTED, SW);
            const char* langs[] = { "Old English", "Spanish" };
            for (int i = 0; i < 2; i++) {
                Rectangle r = RaylibBtn(SW/2, 240 + i*100, 320, 64);
                DrawBtn(r, langs[i], Hov(r), false, C_ACCENT);
                if (Clk(r)) {
                    PlaySound(sClick); ci = i; scr = S_PROF;
                    Speak(std::string(langs[i]) + " selected.");
                }
            }
            Rectangle bk = {30.f, (float)(SH-62), 110.f, 40.f};
            DrawTextClean("< Back", 42, SH-50, 16, Hov(bk) ? C_WHITE : C_MUTED);
            if (Clk(bk)) { PlaySound(sClick); SpeakStop(); scr = S_SPLASH; }
        }

        // ══ PROFICIENCY SELECT ═════════════════
        else if (scr == S_PROF) {
            CText(courses[ci].name.c_str(), 58, 36, C_GOLD, SW);
            CText("Select your Proficiency Level", 106, 20, C_MUTED, SW);

            // Subtitle banner
            const char* profSub[] = {
                "Reading   -  Learn the fundamentals",
                "Writing   -  Build your vocabulary",
                "Translation   -  Master full sentences"
            };

            for (int i = 0; i < (int)courses[ci].profs.size(); i++) {
                Rectangle r = RaylibBtn(SW/2, 180 + i*120, 500, 84);
                bool hov = Hov(r);
                Color pc = ProfColor(i);
                Color bg = hov ? (Color){pc.r/3, pc.g/3, pc.b/3, 255} : C_PANEL;
                DrawRectangleRounded(r, 0.25f, 8, bg);
                DrawRectangleRoundedLines(r, 0.25f, 8, hov ? pc : C_ACCENT);

                // Badge on left
                DrawRectangleRounded({r.x+12, r.y+18, 60.f, 48.f}, 0.3f, 8,
                                     (Color){pc.r/2, pc.g/2, pc.b/2, 200});
                // Roman numeral / icon
                const char* nums[] = {"I","II","III"};
                int nw = (int)MeasureTextEx(g_font, nums[i], 22, 1.0f).x;
                DrawTextClean(nums[i], (int)(r.x+42-nw/2), (int)(r.y+30), 22, pc);

                // Name (big)
                DrawTextClean(courses[ci].profs[i].name.c_str(),
                    (int)(r.x + 88), (int)(r.y + 20), 24, hov ? pc : C_WHITE);
                // Sub label
                DrawTextClean(courses[ci].profs[i].badge.c_str(),
                    (int)(r.x + 90), (int)(r.y + 50), 15, C_MUTED);

                if (Clk(r)) {
                    PlaySound(sClick); pi = i;
                    Speak(courses[ci].profs[i].name + " selected.");
                    startCourse();
                }
            }
            Rectangle bk = {30.f, (float)(SH-62), 110.f, 40.f};
            DrawTextClean("< Back", 42, SH-50, 16, Hov(bk) ? C_WHITE : C_MUTED);
            if (Clk(bk)) { PlaySound(sClick); SpeakStop(); scr = S_LANG; }
        }

        // ══ LEVEL INTRO ════════════════════════
        else if (scr == S_INTRO) {
            const auto& lv = getLevel();
            float fade = fminf(timer * 2.f, 1.f);
            Color fc = {C_WHITE.r, C_WHITE.g, C_WHITE.b, (unsigned char)(255*fade)};
            Color pc = ProfColor(pi);
            // Prof badge top
            char pbadge[64];
            snprintf(pbadge, sizeof(pbadge), "%s  ·  %s",
                     courses[ci].profs[pi].name.c_str(),
                     courses[ci].profs[pi].badge.c_str());
            CText(pbadge, SH/2-110, 18, pc, SW);
            CText(lv.title.c_str(),    SH/2-80, 28, C_MUTED, SW);
            CText(lv.subtitle.c_str(), SH/2-38, 52, fc,      SW);
            DrawLine(SW/2-200, SH/2+28, SW/2+200, SH/2+28, pc);
            CText("Click or press SPACE to begin", SH/2+58, 18, C_MUTED, SW);
            if (timer > 1.0f) introReady = true;
            if (introReady && (IsKeyPressed(KEY_SPACE) || IsMouseButtonPressed(MOUSE_LEFT_BUTTON)))
                goQuiz();
        }

        // ══ QUIZ (Multiple-Choice) ══════════════
        else if (scr == S_QUIZ) {
            const auto& lv = getLevel();
            const auto& q  = getQ();
            Color pc = ProfColor(pi);

            // Header
            char hdr[140];
            snprintf(hdr, sizeof(hdr), "%s  [%s]  -  %s  -  Q%d/%d",
                courses[ci].name.c_str(),
                courses[ci].profs[pi].name.c_str(),
                lv.subtitle.c_str(),
                qi+1, (int)lv.questions.size());
            DrawTextClean(hdr, 28, 14, 12, C_MUTED);
            char sc2[32];
            snprintf(sc2, sizeof(sc2), "Score: %d/%d", score, totalQ);
            int tw = (int)MeasureTextEx(g_font, sc2, 14, 1.0f).x;
            DrawTextClean(sc2, SW - tw - 28, 14, 14, C_GOLD);

            // Proficiency accent line
            DrawRectangle(0, 0, SW, 4, pc);

            // Question panel
            DrawRectangleRounded({28.f, 44.f, (float)(SW-56), 126.f}, 0.15f, 8, C_PANEL);
            DrawTextClean(q.prompt.c_str(), 48, 58, 19, C_WHITE);
            if (!q.subtext.empty())
                CText(q.subtext.c_str(), 96, 26, C_GOLD, SW);

            // Speaker button
            if (DrawSpeaker(SW-88, 58, speaking)) {
                if (speaking) { SpeakStop(); speaking = false; }
                else startSpeak(q.speakText.empty() ? q.prompt : q.speakText);
            }

            // Options
            int oy = 200;
            const int OW = 460, OH = 50;
            for (int i = 0; i < (int)q.options.size(); i++) {
                Rectangle r = RaylibBtn(SW/2, oy, OW, OH);
                bool hov = Hov(r) && !answered;
                Color sc = C_ACCENT;
                if (answered && sel[i]) {
                    bool ok = false;
                    for (int c : q.correct) if (c == i) ok = true;
                    sc = ok ? C_CORRECT : C_WRONG;
                }
                if (answered && !sel[i])
                    for (int c : q.correct)
                        if (c == i) DrawRectangleRoundedLines(r, 0.3f, 8, C_CORRECT);
                std::string lbl = std::to_string(i+1) + ".  " + q.options[i];
                DrawBtn(r, lbl.c_str(), hov, sel[i], sc);
                if (!answered && Clk(r)) { PlaySound(sClick); sel[i] = !sel[i]; }
                oy += OH + 9;
            }

            // Submit button
            if (!answered) {
                Rectangle sb = RaylibBtn(SW/2, oy+12, 210, 48);
                DrawBtn(sb, "Submit", Hov(sb), false, pc);
                if (Clk(sb)) {
                    SpeakStop(); speaking = false;
                    answered = true; correct = true;
                    for (int c : q.correct) if (!sel[c]) { correct = false; break; }
                    if (correct)
                        for (int i = 0; i < (int)q.options.size(); i++)
                            if (sel[i]) {
                                bool v = false;
                                for (int c : q.correct) if (c==i) v=true;
                                if (!v) { correct = false; break; }
                            }
                    if (correct) { score++; PlaySound(sCorrect); Speak("Correct answer!"); }
                    else { PlaySound(sWrong); Speak("Wrong answer. " + q.explanation.substr(0, q.explanation.find('\n'))); }
                    feed = 1.f;
                }
            }
            // Feedback + Next button
            else {
                Color fc2 = correct ? C_CORRECT : C_WRONG;
                CText(correct ? "CORRECT!" : "INCORRECT", oy+4, 22, fc2, SW);
                DrawRectangleRounded({56.f,(float)(oy+32),(float)(SW-112),68.f}, 0.15f, 8, C_PANEL);
                DrawTextClean(q.explanation.c_str(), 74, oy+44, 14, C_MUTED);
                bool last = (qi+1 >= (int)lv.questions.size());
                Rectangle nb = RaylibBtn(SW/2, oy+116, 210, 48);
                DrawBtn(nb, last ? "Next Level >" : "Next Question", Hov(nb), false, pc);
                if (Clk(nb)) { PlaySound(sClick); nextQ(); }
            }

            // Flash on answer
            if (feed > 0.f) {
                Color fl = correct
                    ? (Color){80, 220, 130, (unsigned char)(feed * 50)}
                    : (Color){230, 80,  90,  (unsigned char)(feed * 50)};
                DrawRectangle(0, 0, SW, SH, fl);
            }
        }

        // ══ TYPE-ANSWER QUESTION ════════════════
        else if (scr == S_TYPE) {
            const auto& lv = getLevel();
            const auto& q  = getQ();
            Color pc = ProfColor(pi);

            // Header
            char hdr[140];
            snprintf(hdr, sizeof(hdr), "%s  [%s]  -  %s  -  Q%d/%d",
                courses[ci].name.c_str(),
                courses[ci].profs[pi].name.c_str(),
                lv.subtitle.c_str(),
                qi+1, (int)lv.questions.size());
            DrawTextClean(hdr, 28, 14, 12, C_MUTED);
            char sc2[32];
            snprintf(sc2, sizeof(sc2), "Score: %d/%d", score, totalQ);
            int tw2 = (int)MeasureTextEx(g_font, sc2, 14, 1.0f).x;
            DrawTextClean(sc2, SW - tw2 - 28, 14, 14, C_GOLD);
            DrawRectangle(0, 0, SW, 4, pc);

            // Question panel
            DrawRectangleRounded({28.f, 44.f, (float)(SW-56), 100.f}, 0.15f, 8, C_PANEL);
            DrawTextClean(q.prompt.c_str(), 48, 62, 19, C_WHITE);
            if (!q.subtext.empty())
                CText(q.subtext.c_str(), 112, 22, C_GOLD, SW);

            // Speaker
            if (DrawSpeaker(SW-88, 58, speaking)) {
                if (speaking) { SpeakStop(); speaking = false; }
                else startSpeak(q.speakText.empty() ? q.prompt : q.speakText);
            }

            // Text input box
            int boxy = 180;
            Rectangle inputBox = {80.f, (float)boxy, (float)(SW-160), 52.f};
            Color boxCol = typeSubmitted
                ? (typeCorrect ? C_CORRECT : C_WRONG)
                : pc;
            DrawRectangleRounded(inputBox, 0.2f, 8, C_PANEL);
            DrawRectangleRoundedLines(inputBox, 0.2f, 8, boxCol);
            DrawTextClean(typeInput, 96, boxy+16, 20, C_WHITE);

            // Blinking cursor when not submitted
            if (!typeSubmitted) {
                int cursorX = 96;
                if (g_font.texture.id > 0 && g_font.texture.id != GetFontDefault().texture.id)
                    cursorX += (int)MeasureTextEx(g_font, typeInput, 20, 1.0f).x;
                else
                    cursorX += (int)MeasureTextEx(g_font, typeInput, 20, 1.0f).x;
                if ((int)(timer * 2) % 2 == 0)
                    DrawRectangle(cursorX+2, boxy+12, 2, 26, pc);
            }

            DrawTextClean("Type your answer above and press ENTER to submit", 80, boxy+64, 14, C_MUTED);

            // Handle keyboard input (only when not submitted)
            if (!typeSubmitted) {
                // Backspace
                if (IsKeyPressed(KEY_BACKSPACE) && typeLen > 0) {
                    typeInput[--typeLen] = '\0';
                }
                // Regular chars
                int key = GetCharPressed();
                while (key > 0) {
                    if ((key >= 32) && (typeLen < 250)) {
                        typeInput[typeLen++] = (char)key;
                        typeInput[typeLen]   = '\0';
                    }
                    key = GetCharPressed();
                }
                // Submit on ENTER
                if (IsKeyPressed(KEY_ENTER) && typeLen > 0) {
                    typeSubmitted = true;
                    typeCorrect   = false;
                    std::string ans(typeInput);
                    for (auto& a : q.acceptedAnswers)
                        if (ans == a) { typeCorrect = true; break; }
                    if (typeCorrect) {
                        score++;
                        PlaySound(sCorrect);
                        Speak("Correct answer!");
                    } else {
                        PlaySound(sWrong);
                        Speak("Wrong answer.");
                    }
                    feed = 1.f;
                }
            }

            // Feedback
            if (typeSubmitted) {
                Color fc2 = typeCorrect ? C_CORRECT : C_WRONG;
                CText(typeCorrect ? "CORRECT!" : "INCORRECT", boxy+100, 24, fc2, SW);
                DrawRectangleRounded({56.f,(float)(boxy+134),(float)(SW-112),60.f}, 0.15f, 8, C_PANEL);
                DrawTextClean(q.explanation.c_str(), 74, boxy+146, 14, C_MUTED);

                bool last = (qi+1 >= (int)lv.questions.size());
                Rectangle nb = RaylibBtn(SW/2, boxy+210, 240, 48);
                DrawBtn(nb, last ? "Next Level >" : "Next Question", Hov(nb), false, pc);
                if (Clk(nb)) { PlaySound(sClick); nextQ(); }
            }

            // Flash
            if (feed > 0.f) {
                Color fl = typeCorrect
                    ? (Color){80, 220, 130, (unsigned char)(feed * 50)}
                    : (Color){230, 80,  90,  (unsigned char)(feed * 50)};
                DrawRectangle(0, 0, SW, SH, fl);
            }
        }

        // ══ CONGRATS ═══════════════════════════
        else if (scr == S_CONGRATS) {
            Color pc = ProfColor(pi);
            DrawRectangle(0, 0, SW, 4, pc);

            CText("Proficiency Level Complete!", SH/2-140, 42, C_GOLD, SW);

            // Proficiency name badge
            char pbadge[80];
            snprintf(pbadge, sizeof(pbadge), "%s  —  %s",
                     courses[ci].profs[pi].name.c_str(),
                     courses[ci].profs[pi].badge.c_str());
            CText(pbadge, SH/2-90, 20, pc, SW);

            char sc2[64];
            snprintf(sc2, sizeof(sc2), "Final Score: %d / %d", score, totalQ);
            CText(sc2, SH/2-60, 28, C_WHITE, SW);
            float pct = (totalQ > 0) ? (float)score / totalQ : 0.f;
            CText(pct>=0.8f ? "Outstanding!" : pct>=0.5f ? "Well done!" : "Keep practising!",
                  SH/2-24, 22, C_ACCENT, SW);

            // Score bar
            int bw = 420;
            DrawRectangleRounded({(float)(SW/2-bw/2),(float)(SH/2+8),(float)bw,14.f}, 0.5f, 8, C_PANEL);
            DrawRectangleRounded({(float)(SW/2-bw/2),(float)(SH/2+8),bw*pct,14.f}, 0.5f, 8, pc);

            // Final passage display
            DrawRectangleRounded({40.f,(float)(SH/2+36),(float)(SW-80),128.f}, 0.15f, 8, C_PANEL);
            if (ci == 0) {
                // Old English – Lord's Prayer
                DrawTextClean("THE LORD'S PRAYER (Old English):", 58, SH/2+48, 16, C_GOLD);
                DrawTextClean("F\xC3\xA6der ure, \xC3\xBEu \xC3\xBEe eart on heofonum,", 58, SH/2+68, 14, C_MUTED);
                DrawTextClean("Si \xC3\xBEin nama gehalgod. Tobecume \xC3\xBEin rice.", 58, SH/2+84, 14, C_MUTED);
                DrawTextClean("Our Father, who art in heaven, Hallowed be thy Name.", 58, SH/2+104, 14, C_WHITE);
                DrawTextClean("Thy Kingdom come. Thy will be done in Earth as it is in Heaven.", 58, SH/2+120, 13, C_WHITE);
            } else {
                // Spanish – Final Story
                DrawTextClean("FINAL SPANISH STORY:", 58, SH/2+48, 16, C_GOLD);
                DrawTextClean("Hola amigo. Me llamo Carlos y yo voy a la escuela.", 58, SH/2+68, 14, C_MUTED);
                DrawTextClean("Mi amigo tiene un perro negro y una casa grande.", 58, SH/2+84, 14, C_MUTED);
                DrawTextClean("Hello friend. My name is Carlos and I go to school.", 58, SH/2+104, 14, C_WHITE);
                DrawTextClean("My friend has a black dog and a big house.", 58, SH/2+120, 14, C_WHITE);
            }

            Rectangle ag = RaylibBtn(SW/2-140, SH/2+184, 220, 50);
            Rectangle mn = RaylibBtn(SW/2+140, SH/2+184, 220, 50);
            DrawBtn(ag, "Play Again", Hov(ag), false, pc);
            DrawBtn(mn, "Main Menu",  Hov(mn), false, C_MUTED);
            if (Clk(ag)) { PlaySound(sClick); startCourse(); }
            if (Clk(mn)) { PlaySound(sClick); SpeakStop(); scr = S_LANG; timer = 0.f; }
        }

        EndDrawing();
    }

    // Cleanup
    SpeakStop();
    if (g_font.texture.id > 0 && g_font.texture.id != GetFontDefault().texture.id)
        UnloadFont(g_font);
    UnloadSound(sClick);  UnloadSound(sCorrect);
    UnloadSound(sWrong);  UnloadSound(sIntro);
    CloseAudioDevice();
    CloseWindow();
    return 0;
}
